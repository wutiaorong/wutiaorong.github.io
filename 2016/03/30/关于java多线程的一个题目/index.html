关于java多线程的一个题目

关于java多线程的一道题目

之前网易笔试遇到的一道题目，关于多线程同步的，代码如下：
class myThread extends Thread { int threadid; public myThread(int i) { // TODO Auto-generated constructor stub threadid = i; } public synchronized static void begin(int id) { for (int i = 0; i < 10; i++) { System.out.println("thread: " + id + " " + i); } } @Override public void run() { // TODO Auto-generated method stub begin(threadid); } } public class MyThreadTest { public static void main(String[] args) { for (int i = 0; i < 10; i++) { new myThread(i).start(); try { Thread.sleep(1); } catch (InterruptedException e) { // TODO Auto-generated catch block e.printStackTrace(); } } } }
